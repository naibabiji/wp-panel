function remoteBackup() {
    return {
        cfg: { enabled: false, host: '', port: 22, username: 'root', auth_type: 'key', password: '', remote_path: '', keep_local: true },
        publicKey: '',
        testing: false,
        testResult: '',
        testOK: false,
        saving: false,
        saveMsg: '',
        saveOK: false,
        helperUser: 'backup',
        helperPass: '',
        helperDir: '',
        async fetchRemote() {
            try {
                const r = await api('/settings/remote-backup');
                if (r.success) {
                    Object.assign(this.cfg, r.data);
                    this.helperUser = this.cfg.username || 'backup';
                    this.helperDir = this.cfg.remote_path || '';
                }
                if (this.cfg.auth_type === 'key') this.loadPublicKey();
            } catch(e) {}
        },
        async saveRemote() {
            this.saving = true;
            this.saveMsg = '';
            try {
                const r = await api('/settings/remote-backup', { method: 'PUT', body: this.cfg });
                this.saveOK = r.success;
                this.saveMsg = r.data?.message || (r.success ? '设置已保存' : '保存失败');
                if (r.success && this.cfg.auth_type === 'key') this.loadPublicKey();
            } catch(e) { this.saveOK = false; this.saveMsg = e.message; }
            this.saving = false;
        },
        async loadPublicKey() {
            try {
                const r = await api('/settings/remote-backup');
                if (r.success && r.data.ssh_key) this.publicKey = r.data.ssh_key;
            } catch(e) {}
        },
        async testConnection() {
            this.testing = true;
            this.testResult = '';
            this.testOK = false;
            try {
                const r = await api('/settings/remote-backup/test', { method: 'POST' });
                this.testOK = r.success;
                this.testResult = r.data?.message || (r.success ? '连接成功' : '连接失败');
            } catch(e) { this.testResult = e.message; }
            this.testing = false;
        },
        genPassword() {
            const chars = 'ABCDEFGHJKMNPQRSTUVWXYZabcdefghjkmnpqrstuvwxyz23456789';
            let pw = '';
            for (let i = 0; i < 20; i++) pw += chars[Math.floor(Math.random() * chars.length)];
            this.helperPass = pw;
        },
        genCmd() {
            const u = this.helperUser || 'backup';
            const p = this.helperPass || this.genPassword() || 'changeme';
            const dir = this.helperDir || '/home/' + u + '/backup';
            const authType = this.cfg.auth_type || 'key';

            let cmd = `# === 远程备份账号一键设置 ===
# 在远程服务器以 root 执行以下全部命令：

useradd -m -s /bin/bash ${u}
echo '${u}:${p}' | chpasswd
mkdir -p ${dir}`;

            if (authType === 'key' && this.publicKey) {
                cmd += `
mkdir -p /home/${u}/.ssh
echo '${this.publicKey.trim()}' >> /home/${u}/.ssh/authorized_keys
chmod 700 /home/${u}/.ssh
chmod 600 /home/${u}/.ssh/authorized_keys`;
            }

            cmd += `
chown -R ${u}:${u} ${dir} /home/${u}
chmod 755 ${dir}
echo '账号 ${u} 创建完成，备份目录：${dir}'`;

            return cmd;
        },
        async copyCmd() {
            const cmd = this.genCmd();
            if (!cmd) return;
            try {
                await navigator.clipboard.writeText(cmd);
                showToast('命令已复制，粘贴到远程服务器以 root 执行', 'success');
            } catch(e) {
                const ta = document.createElement('textarea');
                ta.value = cmd;
                document.body.appendChild(ta);
                ta.select();
                document.execCommand('copy');
                document.body.removeChild(ta);
                showToast('命令已复制', 'success');
            }
        }
    }
}
